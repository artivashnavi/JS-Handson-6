var Perfect_Check = (N) => 
{
     sum = 1;
 
    
    for (let i=2; i*i<=N; i++)
    {
        if (N%i===0)
        {
            if(i*i!=N)
                sum = sum + i + N/i;
            else
                sum=sum+i;
        }
    }
   
    if (sum === N && N !== 1)
